                \item html页面等已有的内容->直接发过去
                \item 类比printf("Hello world!");
            \end{itemize}
            \item 动态内容\begin{itemize}
                \item 需要运行程序才能知道发送什么->fork一下并运行
                \item 类比printf("\%ld",time());
            \end{itemize}
            \item web内容与服务器上的文件相联系
        \end{itemize}
